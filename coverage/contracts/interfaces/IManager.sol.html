pragma solidity 0.8.0;
 
import './IKeeperFees.sol';
import './ISanToken.sol';
import './IVaultManager.sol';
import './IOracle.sol';
 
/// @title IManager
/// @author Angle Core Team
/// @notice Interface for token Managers.
interface IManager {
 
	// ============================ Constructor ===================================
 
	function deployCollateral(
		address governor,
		address guardian,
		ISanToken _sanToken,
		IVaultManager _vaultManager,
		IKeeperFees keeper
	) external;
 
	// ============================ Users / SLP ===============================
 
	function exit(address user, uint256 amount) external;
 
	// ============================ Governance ===================================
 
	function deployLendingInterface(
		bytes32 platformName,
		address _contractAddress,
		uint256[] memory bounds
	) external;
 
	function revokeLendingInterface(bytes32 platformName) external;
 
	function setGovernor(address _governor, address governor) external;
 
	function setGuardian(address _guardian, address guardian) external;
 
	function revokeGuardian(address guardian) external;
 
	function transferToSettlement(address settlementContract,uint256 amount) external;
 
	function setStableMaster(IStableMaster _stableMaster) external;
 
	function setOracle(IOracle _oracle) external;
 
	function setBase(uint256 _BASE) external;
 
	function setLendingBounds(bytes32 platformName, uint256[] memory _bounds) external;
 
	function setKeeperFeesLiquidate(uint256 _keeperFeesLiquidate) external;
 
	function setKeeperFeesCashOutCap(uint256 _keeperFeesCashOutCap) external;
 
	function setXKeeperFeesCashOut(uint256[] memory _XKeeperFeesCashOut) external;
 
	function setYKeeperFeesCashOut(uint256[] memory _YKeeperFeesCashOut) external;
 
	function pauseStabler() external;
 
	function unpauseStabler() external;
 
	function pauseHA() external;
 
	function unpauseHA() external;
 
	function pauseSLP() external;
 
	function unpauseSLP() external;
 
	function withdrawAllLending() external;
 
	// ============================ Getters ===================================
 
	function getBalance() external view returns (uint256);
 
	function getTotalAsset() external view returns (uint256);
}

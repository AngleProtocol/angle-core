        IERC20Metadata outCur = IERC20Metadata(circuitAddressERC20[0]);

    /// @notice updateall rates from the circuit of both Uniswap and Chainlink
    /// @return whether the tx completed or not.
    function updateAll() external returns (bool) {
        uint32[] memory secondAgos = new uint32[](2);
        secondAgos[0] = twapPeriod;
        secondAgos[1] = 0;
        int56 tickFinal;
        int56 tickCumulativesDelta;
        int56[] memory tickCumulatives;
        int24 timeWeightedAverageTick;
 
        // We are doing the computation on ticks first and then convert tick to a price for I guess more precision
        // Iw we do P1*P2/P3 this is the same as computing 1.0001**(t1+t2-t3)
        uint256 rateUniswap;
        for (uint256 i = 0; i<circuitUniswap.length;i++){
            // Adapted fro uniswap v3 uniswap-v3-periphery/contracts/libraries/oracleLibrary.sol
            // Fetches time-weighted average tick using Uniswap V3 oracle
            // timeWeightedAverageTick The time-weighted average tick from (block.timestamp - period) to block.timestamp
            (tickCumulatives, ) = circuitUniswap[i].observe(secondAgos);
            tickCumulativesDelta = tickCumulatives[1] - tickCumulatives[0];
 
            timeWeightedAverageTick = int24(tickCumulativesDelta / twapPeriod);
 
            // Always round to negative infinity
            if (tickCumulativesDelta < 0 && (tickCumulativesDelta % twapPeriod != 0)) timeWeightedAverageTick--;
 
            if(circuitUniIsMulitplied[i]==1) tickFinal += timeWeightedAverageTick;
            else tickFinal -= timeWeightedAverageTick;
        }
 
        uint256 rateChainLink;
        for (uint256 i = 0; i<circuitChainlink.length;i++){
            (,int ratio,,,) = circuitChainlink[i].latestRoundData();
        }
 
        return true;
    }

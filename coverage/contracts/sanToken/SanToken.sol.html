pragma solidity 0.8.0;
 
import '@openzeppelin/contracts/security/Pausable.sol';
import '@openzeppelin/contracts/token/ERC20/extensions/ERC20Burnable.sol';
import '@openzeppelin/contracts-upgradeable/proxy/utils/Initializable.sol';
 
import '../interfaces/ISanToken.sol';
import '../interfaces/IManager.sol';
 
/// @title Base contract for sanTokens
/// @author Angle Core Team
/// @notice These tokens are used to mark the debt to SLPs
contract SanToken is ISanToken, ERC20Burnable, AccessControl, Initializable, Pausable {
	bytes32 public constant STABLEMASTER_ROLE = keccak256('STABLEMASTER_ROLE');
	bytes32 public constant MANAGER_ROLE = keccak256('MANAGER_ROLE');
	bytes32 public constant GOVERNOR_ROLE = keccak256('GOVERNOR_ROLE');
	bytes32 public constant GUARDIAN_ROLE = keccak256('GUARDIAN_ROLE');
 
	/// @notice address of the corresponding collateral manager
	IManager public manager;
 
	/// @notice address of the corresponding stableMaster contract
	IStableMaster public stableMaster;
 
	// ========================= Constructor ===================
 
	/// @notice sanToken constructor
	/// @param _name of the token
	/// @param _symbol of the token
	constructor(
		string memory _name,
		string memory _symbol,
		IStableMaster _stableMaster,
		IManager _manager
	) ERC20(_name, _symbol) {
		manager = _manager;
		stableMaster = _stableMaster;
 
		_setupRole(STABLEMASTER_ROLE, address(_stableMaster));
		_setupRole(MANAGER_ROLE, address(_manager));
 
		_setRoleAdmin(STABLEMASTER_ROLE, MANAGER_ROLE);
		_setRoleAdmin(MANAGER_ROLE, MANAGER_ROLE);
		_setRoleAdmin(GOVERNOR_ROLE, MANAGER_ROLE);
		_setRoleAdmin(GUARDIAN_ROLE, MANAGER_ROLE);
	}
 
	/// @notice initiate the gover and guardian role
	/// @param governor address of the protocol
	/// @param guardian address
	function init(address governor, address guardian) external override onlyRole(MANAGER_ROLE) initializer {
		grantRole(GOVERNOR_ROLE, governor);
		grantRole(GUARDIAN_ROLE, governor);
		grantRole(GOVERNOR_ROLE, address(stableMaster));
		grantRole(GUARDIAN_ROLE, address(stableMaster));
		grantRole(GUARDIAN_ROLE, guardian);
	}
 
	// ========================= External Functions =======================
 
	/// @notice to burn this sanToken
	/// @param account to burn from
	/// @param amount to burn
	function burnFrom(address account, uint256 amount) public override whenNotPaused {
		uint256 currentAllowance = allowance(account, _msgSender());
		require(currentAllowance >= amount, 'sanToken::burnFrom: Burn amount exceeds allowance');
		_approve(account, _msgSender(), currentAllowance - amount);
		_burn(account, amount);
 
		stableMaster.notifyWithdraw(amount, account);
	}
 
	/// @notice to burn this stablecoin
	/// @param amount to burn from caller
	function burnSelf(uint256 amount) external whenNotPaused {
		_burn(msg.sender, amount);
		stableMaster.notifyWithdraw(amount, msg.sender);
	}
 
	// ============================ GOVERNANCE ==========================
 
	// ============================ Manager Functions ==========================
 
	/// @notice set a new stable master contract
	/// @param _stableMaster contract
	function setStableMaster(IStableMaster _stableMaster) external override onlyRole(MANAGER_ROLE) {
		grantRole(STABLEMASTER_ROLE, address(_stableMaster));
		grantRole(GOVERNOR_ROLE, address(_stableMaster));
		grantRole(GUARDIAN_ROLE, address(_stableMaster));
 
		revokeRole(GOVERNOR_ROLE, address(stableMaster));
		revokeRole(GUARDIAN_ROLE, address(stableMaster));
		revokeRole(STABLEMASTER_ROLE, address(stableMaster));
		stableMaster = _stableMaster;
	}
 
	// ============================ StableMaster Functions ==========================
 
	/// @notice to mint this token
	/// @param account to mint to
	/// @param amount to mint
	function mint(address account, uint256 amount) external override onlyRole(STABLEMASTER_ROLE) {
		_mint(account, amount);
	}
 
	// TODO what's the role of governor here, no function uses it
 
	// ============================ Guardian Functions ==========================
 
	/// @notice set pausable methods to paused
	function pause() external override onlyRole(GUARDIAN_ROLE) {
		_pause();
	}
 
	/// @notice set pausable methods to unpaused
	function unpause() external override onlyRole(GUARDIAN_ROLE) {
		_unpause();
	}
 
 
}

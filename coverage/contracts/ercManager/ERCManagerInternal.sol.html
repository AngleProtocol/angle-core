pragma solidity 0.8.0;
 
import './ERCManagerStorage.sol';
 
/// @title ERCManager
/// @author Angle Core Team
/// @notice Manager of the ERC tokens.
/// @dev Internal functions only
contract ERCManagerInternal is ERCManagerStorage {
 
	// ============================ Lending ===================================
 
	/// @notice Re-establish the lending ratio set by governance
	/// @param platformName the lending protocol name
	function _correctLendingRate(bytes32 platformName) internal returns (uint256 gain) {
		LendingInterface storage platform = lendingInterfaceMap[platformName];
		// No need to test if the platform exists well in this case
		uint256 totAssetsInC = _getTotalAsset();
		require(totAssetsInC != 0, 'ERCManager::_correctLendingRate: No assets to manage');
 
		uint256 currentLendingRatio = (platform.amountLent * BASE) / totAssetsInC;
		if (currentLendingRatio < platform.lowerBound) {
			uint256 toLend = ((platform.target - currentLendingRatio) * totAssetsInC) / BASE;
			gain = platform.interfaceContract.depositLending(toLend);
			platform.amountLent = platform.amountLent + toLend;
		} else if (currentLendingRatio > platform.upperBound) {
			uint256 toWithdraw = ((currentLendingRatio - platform.target) * totAssetsInC) / BASE;
			gain = platform.interfaceContract.withdrawLending(toWithdraw);
			platform.amountLent = platform.amountLent - toWithdraw;
		}
	}
 
	/// @notice Distributes the lending gains made accross all lending platforms
	function _distributeLendingGains() internal {
		uint256 value = 0;
		for (uint256 i = 0; i < lendingInterfacesList.length; i++) {
			value = value + getLendingGains(lendingInterfacesList[i]);
		}
		if (value > 0) {
			stableMaster.accumulateFees(value);
			emit DistributingFees(value);
		}
	}
 
	/// @notice Distributes the lending gains made accross a lending platform
	/// @param platformName the lending protocol name
	function _distributeLendingGains(bytes32 platformName) internal {
		uint256 value = getLendingGains(platformName);
		if (value > 0) {
			stableMaster.accumulateFees(value);
			emit DistributingFees(value);
		}
	}
 
	/// @notice compute the earnings from lending protocol (name) between the last call and now
	/// @param platformName the lending protocol name
	function getLendingGains(bytes32 platformName) internal returns (uint256) {
		LendingInterface storage platform = lendingInterfaceMap[platformName];
		if (platform.amountLent == 0) {
			return 0;
		} else {
			uint256 lendingGains = platform.interfaceContract.getLendingGains();
			platform.amountLent = platform.amountLent + lendingGains;
			return lendingGains;
		}
	}
 
	// ============================ Utils ===================================
 
	/// @notice returns the manager reserve of collateral
	function _getBalance() internal view returns (uint256) {
		return token.balanceOf(address(this));
	}
 
	/// @notice Allow to transfer amount to owner and if not enough reserve then the HA
	/// receive sanTokens
	/// @param owner address of the receiver
	/// @param amount the amount of collateral sent
	function secureTransfer(address owner, uint256 amount) internal returns (bool) {
		uint256 curBalance = _getBalance();
		if (curBalance >= amount) {
			require(token.transfer(owner, amount), 'ERCManager::secureTransfer: Failed transfer');
			return true;
		} else {
			// not enough to reimburse the entire amount so we transform part of her position to a SLP
			uint256 amountLeft = curBalance - amount;
			require(token.transfer(owner, curBalance), 'ERCManager::secureTransfer: Failed transfer');
			stableMaster.notifyDeposit(amountLeft, owner);
			return true;
		}
	}
 
	/// @notice return the number of owned assets by the manager
	function _getTotalAsset() internal view returns (uint256) {
		uint256 reserves = _getBalance();
		for (uint256 i = 0; i < lendingInterfacesList.length; i++) {
			reserves = reserves + lendingInterfaceMap[lendingInterfacesList[i]].amountLent;
		}
		return reserves;
	}
}

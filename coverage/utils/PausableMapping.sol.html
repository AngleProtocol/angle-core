// SPDX-License-Identifier: UNLICENSED
 
pragma solidity ^0.8.0;
 
import "@openzeppelin/contracts/utils/Context.sol";
 
/**
 * @dev Contract module which allows children to implement an emergency stop
 * mechanism that can be triggered by an authorized account.
 *
 * This module is used through inheritance. It will make available the
 * modifiers `whenNotPaused` and `whenPaused`, which can be applied to
 * the functions of your contract. Note that they will not be pausable by
 * simply including this module, only once the modifiers are put in place.
 */
abstract contract PausableMap is Context {
    /**
     * @dev Emitted when the pause is triggered by `account`.
     */
    event Paused(string name, address account);
 
    /**
     * @dev Emitted when the pause is lifted by `account`.
     */
    event Unpaused(string name, address account);
 
    /**
     * @dev mapping of paused bool representing the state.
     */
    mapping(string => bool) private _paused;
 
    /**
     * @dev Initializes the contract in unpaused state.
     */
    constructor() {}
 
    /**
     * @dev Returns true if the contract is paused for stable holders , and false otherwise.
     */
    function paused(string memory name) public view virtual returns (bool) {
        return _paused[name];
    }
 
    /**
     * @dev Modifier to make a function callable only when the contract is not paused for stablers.
     *
     * Requirements:
     *
     * - The contract must not be paused for stablers.
     */
    modifier whenNotPaused(string memory name) {
        require(!paused(name), "Pausable: paused");
        _;
    }
 
    /**
     * @dev Modifier to make a function callable only when the contract is paused for stablers.
     *
     * Requirements:
     *
     * - The contract must be paused for Stablers.
     */
    modifier whenPaused(string memory name) {
        require(paused(name), "Pausable: not paused");
        _;
    }
 
    /**
     * @dev Triggers stopped state.
     *
     * Requirements:
     *
     * - The contract must not be paused.
     */
    function _pause(string memory name) internal virtual whenNotPaused(name) {
        _paused[name] = true;
        emit Paused(name, _msgSender());
    }
 
    /**
     * @dev Returns to normal state.
     *
     * Requirements:
     *
     * - The contract must be paused.
     */
    function _unpause(string memory name) internal virtual whenPaused(name) {
        _paused[name] = false;
        emit Unpaused(name, _msgSender());
    }
}

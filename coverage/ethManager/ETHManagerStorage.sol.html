pragma solidity 0.8.0;
 
import {IERC20} from '@openzeppelin/contracts/token/ERC20/IERC20.sol';
import {SafeERC20} from '@openzeppelin/contracts/token/ERC20/utils/SafeERC20.sol';
 
import '../external/FunctionUtils.sol';
import '../external/PausableAgents.sol';
 
import '../interfaces/IManager.sol';
import '../interfaces/ILending.sol';
import '../interfaces/IVaultManager.sol';
import '../interfaces/IStableMaster.sol';
import '../interfaces/ISanToken.sol';
 
/// @title Manager for ETH
/// @author Angle Core Team
/// @notice Manager of the ETH used as collateral
contract ETHManagerStorage {
	//TODO bounds need to hardcode coherent bounds such that there is not
	//     more than 100% lent or other issues
	struct LendingInterface {
		ILending interfaceContract;
		uint256 lowerBound;
		uint256 upperBound;
		uint256 target;
		uint256 amountLent;
	}
 
	string public name;
	IStableMaster public stableMaster;
	ISanToken public sanToken;
	IVaultManager public vaultManager;
	IKeeperFees public keeper;
 
	// this one needs to be the same for all contracts
	uint256 public BASE;
 
	// parameters for keeper fees
	uint256 keeperFeesLiquidate;
	// TODO change the cap
	uint256 keeperFeesCashOutCap;
	// these are only multiplier of |gap leverage - committed|
	uint256[] XKeeperFeesCashOut;
	uint256[] YKeeperFeesCashOut;
 
	mapping(bytes32 => LendingInterface) public lendingInterfaceMap;
	bytes32[] public lendingInterfacesList;
 
	// Events we can't place in IManager.sol because they concern internal functions that do not implement this class
	// TODO lots of duplicates between ETH and ERC Manager: we should think about merging some files and creating inheritances
	event Lending(bytes32 platformName, uint256 amount);
 
	event Withdrawing(bytes32 platformName, uint256 amount);
 
	event UpdatingExchangeRate(bytes32 platformName, uint256 newExchangeRate);
 
	event DistributingFees(uint256 amountDistributed);
 
	// ============================ Constructor ===================================
 
	/// @notice Constructor
	/// @param _name of the manager vault (collateral)
	/// @param _stableMaster interface of the stable master of this manager
	function initializeParameters(string memory _name, IStableMaster _stableMaster) internal {
		name = _name;
		stableMaster = _stableMaster;
 
		BASE = 10**18;
		keeperFeesLiquidate = (5 * BASE) / 10;
		keeperFeesCashOutCap = BASE**2;
		XKeeperFeesCashOut = [BASE / 2, BASE, 2 * BASE];
		YKeeperFeesCashOut = [BASE / 10, (6 * BASE) / 10, BASE / 10];
	}
}

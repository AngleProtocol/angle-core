// @notice mock contract to lend ERC20 tokens to Compound
contract MockERC20Compound is CErc20Compound, ERC20 {
	using SafeERC20 for IERC20;
 
	IERC20 public token; // Interface for the token
	event Minting(address indexed _to, address indexed _minter, uint256 _amount);
 
	event Burning(address indexed _from, address indexed _burner, uint256 _amount);
	uint256 BASE = 10**9;
	uint256 public exchangeRateStored_ = 10**9;
 
	/// @notice stablecoin constructor
	/// @param name_ of the token lent
	/// @param symbol_ of the token lent
	constructor(
		string memory name_,
		string memory symbol_,
		address token_
	) ERC20(name_, symbol_) {
		token = IERC20(token_);
	}
 
	function mint(uint256 amount) external override returns (uint256) {
		token.safeTransferFrom(msg.sender, address(this), amount);
		uint256 amount_ctoken = (amount * BASE) / exchangeRateStored_;
		_mint(msg.sender, amount_ctoken);
		emit Minting(msg.sender, msg.sender, amount_ctoken);
		// you need to transfer first between the person and the other the token involved
		return (amount_ctoken);
		//emit Minting(account, msg.sender, amount);
	}
 
	function changeExchangeRateStored(uint256 newExchangeRate) external {
		exchangeRateStored_ = newExchangeRate * BASE;
	}
 
	function redeem(uint256 amount) external returns (uint256) {
		burn(msg.sender, amount);
		uint256 amount_token = (amount * exchangeRateStored_) / BASE;
		token.safeTransfer(msg.sender, amount_token);
		return (amount_token);
		//emit Minting(account, msg.sender, amount);
	}
 
	function redeemUnderlying(uint256 amount) external override returns (uint256) {
		uint256 amount_cToken = (amount * BASE) / exchangeRateStored_;
		burn(msg.sender, amount_cToken);
		uint256 amount_token = (amount_cToken * exchangeRateStored_) / BASE;
		token.safeTransfer(msg.sender, amount_token);
		return (amount_token);
	}
 
	function balanceOfUnderlying(address owner) external view override returns (uint256) {
		return (this.balanceOf(owner) * exchangeRateStored_) / BASE;
	}
 
	/*
	function exchangeRateStored() external view override returns (uint256) {
		return exchangeRateStored_.div(BASE);
	} */
 
	function mint(address account, uint256 amount) external {
		_mint(account, amount);
		emit Minting(account, msg.sender, amount);
	}
 
	function burn(address account, uint256 amount) public {
		_burn(account, amount);
		emit Burning(account, msg.sender, amount);
	}
}

pragma solidity 0.8.0;
 
import '@openzeppelin/contracts/token/ERC20/utils/SafeERC20.sol';
import '@openzeppelin/contracts/token/ERC20/IERC20.sol';
 
import '../external/LendingInterfaces.sol';
 
import '../interfaces/ILending.sol';
 
/// @title LendingCompound
/// @author Angle Core Team
/// @notice Contract interacting with Compound to lend a specific type of ERC20 token
/// There will be one such contract per ERC20 token
contract LendingCompoundERC20 is ILending, AccessControl {
	using SafeERC20 for IERC20;
 
	bytes32 public constant MANAGER_ROLE = keccak256('MANAGER_ROLE');
	bytes32 public constant GUARDIAN_ROLE = keccak256('GUARDIAN_ROLE');
 
	CErc20Compound public cToken;
 
	IERC20 public token;
 
	address public manager;
 
	uint256 public balanceOfUnderlying = 0;
 
	// ============================ Constructor ===================================
 
	/// @notice Constructor
	/// @param _cToken Interface of Compound's contract
	/// @param _token address of the underlying token
	/// @param _manager address of the corresponding ERCManager
	/// @param governor address of the governor
	/// @param guardian address of the guardian
	constructor(
		CErc20Compound _cToken,
		IERC20 _token,
		address _manager,
		address governor,
		address guardian
	) {
		cToken = _cToken;
		token = _token;
		manager = _manager;
		_setupRole(MANAGER_ROLE, _manager);
		_setupRole(GUARDIAN_ROLE, governor);
		_setupRole(GUARDIAN_ROLE, guardian);
		_setRoleAdmin(MANAGER_ROLE, MANAGER_ROLE);
		_setRoleAdmin(GUARDIAN_ROLE, MANAGER_ROLE);
 
		// As this contract will not have any stock, approve the lending contract
		token.approve(address(cToken), 2**256 - 1);
	}
 
	// ============================ FUNCTIONS ===================================
 
	/// @notice Deposits ERC20 on Compound
	/// @param amountToLend on the protocol
	function depositLending(uint256 amountToLend) external override onlyRole(MANAGER_ROLE) returns (uint256 gain) {
		uint256 newBalanceOfUnderlying = cToken.balanceOfUnderlying(address(this));
		gain = newBalanceOfUnderlying - balanceOfUnderlying;
		emit LendingGainsAchieved(gain);
 
		// TODO we should maybe put an approve here instead of doing it in the constructor
		token.safeTransferFrom(manager, address(this), amountToLend);
		cToken.mint(amountToLend);
		emit Lending(amountToLend);
 
		balanceOfUnderlying = cToken.balanceOfUnderlying(address(this));
	}
 
	/// @notice Withdraws ERC20 from Compound
	/// @param amountToWithdraw from the lending protocol
	function withdrawLending(uint256 amountToWithdraw) external override onlyRole(MANAGER_ROLE) returns (uint256 gain) {
		uint256 newBalanceOfUnderlying = cToken.balanceOfUnderlying(address(this));
		gain = newBalanceOfUnderlying - balanceOfUnderlying;
		emit LendingGainsAchieved(gain);
 
		// redeem type is always true in this interface, we redeem the asset based on an amount of it
		// Retrieve your asset based on an amount of the asset
		require(newBalanceOfUnderlying > amountToWithdraw, 'LendingCompoundERC20:withdrawLending Incorrect balance');
 
		// redeem type is always true in this interface, we redeem the asset based on an amount of it
		// Retrieve your asset based on an amount of the asset
		cToken.redeemUnderlying(amountToWithdraw);
		token.safeTransfer(manager, token.balanceOf(address(this)));
		emit Withdrawing(amountToWithdraw);
 
		balanceOfUnderlying = cToken.balanceOfUnderlying(address(this));
	}
 
	/// @notice Computes the capital gains made by lending ERC20 on Compound
	function getLendingGains() external override onlyRole(MANAGER_ROLE) returns (uint256 gain) {
		uint256 newBalanceOfUnderlying = cToken.balanceOfUnderlying(address(this));
		gain = newBalanceOfUnderlying - balanceOfUnderlying;
		emit LendingGainsAchieved(gain);
		balanceOfUnderlying = newBalanceOfUnderlying;
	}
 
	/// @notice In case of, changes the approval amount for Compound contract
	function changeTokenApprovalAmount(uint256 newAmount) external onlyRole(GUARDIAN_ROLE) {
		token.approve(address(cToken), newAmount);
	}
}

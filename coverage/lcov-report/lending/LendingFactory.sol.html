pragma solidity 0.8.0;
pragma experimental ABIEncoderV2;
 
import './LendingAaveERC20.sol';
import './LendingCompoundERC20.sol';
import './LendingCompoundETH.sol';
 
contract LendingFactory {
	constructor() {}
 
	address public lastDeployed;
 
	function newLendingAaveERC20(
		AErc20Aave _lendingPool,
		IERC20 _token,
		IERC20 _aaveToken,
		IManager _manager,
		address governor,
		address guardian
	) public {
		LendingAaveERC20 len = (new LendingAaveERC20(_lendingPool, _token, _aaveToken, _manager, governor, guardian));
		lastDeployed = address(len);
	}
 
	function newLendingCompoundERC20(
		CErc20Compound _cToken,
		IERC20 _token,
		address _manager,
		address governor,
		address guardian
	) public {
		LendingCompoundERC20 len = (new LendingCompoundERC20(_cToken, _token, _manager, governor, guardian));
		lastDeployed = address(len);
	}
 
	function newLendingCompoundETH(
		address compoundContract,
		address _manager,
		address governor,
		address guardian
	) public {
		LendingCompoundETH len = (new LendingCompoundETH(compoundContract, _manager, governor, guardian));
		lastDeployed = address(len);
	}
}

// SPDX-License-Identifier: UNLICENSED
 
pragma solidity ^0.8.0;
 
import "@openzeppelin/contracts-upgradeable/access/AccessControlUpgradeable.sol";
 
/**
 * @dev Contract module which allows children to implement an emergency stop
 * mechanism that can be triggered by an authorized account.
 *
 * This module is used through inheritance. It will make available the
 * modifiers `whenNotPaused` and `whenPaused`, which can be applied to
 * the functions of your contract. Note that they will not be pausable by
 * simply including this module, only once the modifiers are put in place.
 */
abstract contract PausableAgents is AccessControlUpgradeable {
    /**
     * @dev Emitted when the pause is triggered by `account`.
     */
    event PausedStabler(address account);
    event PausedHA(address account);
    event PausedSLP(address account);
 
    /**
     * @dev Emitted when the pause is lifted by `account`.
     */
    event UnpausedStabler(address account);
    event UnpausedHA(address account);
    event UnpausedSLP(address account);
 
    bool private _pausedStabler;
 
    bool private _pausedHA;
 
    bool private _pausedSLP;
 
    /**
     * @dev Initializes the contract in unpaused state.
     */
    function initializePausableAgents() public initializer {
        __AccessControl_init();
        _pausedStabler = false;
        _pausedHA = false;
        _pausedSLP = false;
    }
 
    /**
     * @dev Returns true if the contract is paused for stable holders , and false otherwise.
     */
    function pausedStabler() public view virtual returns (bool) {
        return _pausedStabler;
    }
 
    /**
     * @dev Returns true if the contract is paused fro HA, and false otherwise.
     */
    function pausedHA() public view virtual returns (bool) {
        return _pausedHA;
    }
 
    /**
     * @dev Returns true if the contract is paused fro SLP, and false otherwise.
     */
    function pausedSLP() public view virtual returns (bool) {
        return _pausedSLP;
    }
 
    /**
     * @dev Modifier to make a function callable only when the contract is not paused for stablers.
     *
     * Requirements:
     *
     * - The contract must not be paused for stablers.
     */
    modifier whenNotPausedStabler() {
        require(!pausedStabler(), "Pausable Stabler: paused");
        _;
    }
 
    /**
     * @dev Modifier to make a function callable only when the contract is paused for stablers.
     *
     * Requirements:
     *
     * - The contract must be paused for Stablers.
     */
    modifier whenPausedStabler() {
        require(pausedStabler(), "Pausable Stabler: not paused");
        _;
    }
 
    /**
     * @dev Modifier to make a function callable only when the contract is not paused for HA.
     *
     * Requirements:
     *
     * - The contract must not be paused for stablers.
     */
    modifier whenNotPausedHA() {
        require(!pausedHA(), "Pausable HA: paused");
        _;
    }
 
    /**
     * @dev Modifier to make a function callable only when the contract is paused for HA.
     *
     * Requirements:
     *
     * - The contract must be paused for Stablers.
     */
    modifier whenPausedHA() {
        require(pausedHA(), "Pausable HA: not paused");
        _;
    }
 
    /**
     * @dev Modifier to make a function callable only when the contract is not paused for SLP.
     *
     * Requirements:
     *
     * - The contract must not be paused for stablers.
     */
    modifier whenNotPausedSLP() {
        require(!pausedSLP(), "Pausable SLP: paused");
        _;
    }
 
    /**
     * @dev Modifier to make a function callable only when the contract is paused for SLP.
     *
     * Requirements:
     *
     * - The contract must be paused for Stablers.
     */
    modifier whenPausedSLP() {
        require(pausedSLP(), "Pausable SLP: not paused");
        _;
    }
 
    /**
     * @dev Triggers stopped state for stabler.
     *
     * Requirements:
     *
     * - The contract must not be paused for stablers.
     */
    function _pauseStabler() internal virtual whenNotPausedStabler {
        _pausedStabler = true;
        emit PausedStabler(_msgSender());
    }
 
    /**
     * @dev Triggers stopped state for ha.
     *
     * Requirements:
     *
     * - The contract must not be paused for ha.
     */
    function _pauseHA() internal virtual whenNotPausedHA {
        _pausedHA = true;
        emit PausedHA(_msgSender());
    }
 
    /**
     * @dev Triggers stopped state for SLP .
     *
     * Requirements:
     *
     * - The contract must not be paused for SLP.
     */
    function _pauseSLP() internal virtual whenNotPausedSLP {
        _pausedSLP = true;
        emit PausedSLP(_msgSender());
    }
 
    /**
     * @dev Returns to normal state for Stabler.
     *
     * Requirements:
     *
     * - The contract must be paused fdor Stabler.
     */
    function _unpauseStabler() internal virtual whenPausedStabler {
        _pausedStabler = false;
        emit UnpausedStabler(_msgSender());
    }
 
    /**
     * @dev Returns to normal state for HA.
     *
     * Requirements:
     *
     * - The contract must be paused for HA.
     */
    function _unpauseHA() internal virtual whenPausedHA {
        _pausedHA = false;
        emit UnpausedHA(_msgSender());
    }
 
    /**
     * @dev Returns to normal state for SLP.
     *
     * Requirements:
     *
     * - The contract must be paused for SLP.
     */
    function _unpauseSLP() internal virtual whenPausedSLP {
        _pausedSLP = false;
        emit UnpausedSLP(_msgSender());
    }
}

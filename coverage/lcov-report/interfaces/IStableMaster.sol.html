        address governor,
        address guardian,
        address _aToken
    ) external;
 
    // ============================== Users ======================================
 
    function notifyMint(uint256 amount, address user) external;
 
    function notifyBurn(
        uint256 amount,
        IManager manager,
        address user

    function notifyDeposit(uint256 amount, address user) external;
 
    function notifyWithdraw(uint256 amount, address user) external;

    function getStockUsers() external view returns (int256);
 
    function updateStockUsers(int256 amount) external;

    function getMint() external view returns (uint256);

    function accumulateFees(uint256 fees) external;
 
    // ============================== Keepers =====================================
 
    function getCollateralRatio() external returns (uint256);

        uint256 feeMint,
        uint256 feeBurn,
        uint256 _slippage,

    function setStableMaster(IStableMaster _stableMaster) external;
 
    function setKeeper(address newKeeper, address oldKeeper) external;
 
    function setOracle(IOracle _oracle, IManager manager) external;
 
    function setBaseGlobal(uint256 _base) external;
 
    function updateStockUsersGov(int256 amount, IManager manager) external;
 
    function setFeesForLPs(uint256 _feesForLPs) external;
 
    function setMaxALock(uint256 _maxALock) external;
 
    function setFeeMint(
        IManager manager,
        uint256[] memory _xFeeMint,
        uint256[] memory _yFeeMint
    ) external;
}

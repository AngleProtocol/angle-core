/// @title Interface of the contract managing vaults
/// @dev Front interface, meaning only user-facing functions
/// @author Angle Core Team
interface IVaultManagerFront {
    // ========================= Gov Token Distribution =============================
 
    function getReward(uint256 vaultID) external;
 
    function earned(uint256 vaultID) external view returns (uint256);
 
    // ========================= ERC721 =============================
 
    function balanceOf(address owner) external view returns (uint256 balance);
 
    function ownerOf(uint256 vaultID) external view returns (address owner);
 
    function safeTransferFrom(
        address from,
        address to,
        uint256 vaultID
    ) external;
 
    function transferFrom(
        address from,
        address to,
        uint256 vaultID
    ) external;
 
    function approve(address to, uint256 vaultID) external;
 
    function getApproved(uint256 vaultID) external view returns (address operator);
 
    function setApprovalForAll(address operator, bool _approved) external;
 
    function isApprovedForAll(address owner, address operator) external view returns (bool);
 
    function safeTransferFrom(
        address from,
        address to,
        uint256 vaultID,
        bytes calldata data
    ) external;
}
 
/// @title Interface of the contract managing vaults
/// @author Angle Core Team
interface IVaultManager is IAccessControlUpgradeable, IERC165Upgradeable {
    // ========================= Governance ====================
 
    function deployCollateral(
        address governor,
        address guardian,
        IKeeperFees keeper
    ) external;
 
    // temporary for testing the contract
    function setManagerRole(address tester) external;
 
    function setStableMaster(IStableMaster _stableMaster) external;
 
    function setKeeper(IKeeperFees _keeperFees) external;
 
    function setOracle(IOracle _oracle) external;
 
    function setBase(uint256 _base) external;
 
    function setSecureBlocks(uint256 _secureBlocks) external;
 
    function setMaxLeverage(uint256 maxLeverageNumerator, uint256 maxLeverageDenominator) external;
 
    function setMurDistribution(uint256 _rewardsDuration, address _rewardsDistribution) external;
 
    function deactivateMurDistribution() external;
 
    function setXHAFeesDeposit(uint256[] memory _xHAFeesDeposit) external;
 
    function setYHAFeesDeposit(uint256[] memory _yHAFeesDeposit) external;
 
    function setXHAFeesWithdraw(uint256[] memory _xHAFeesWithdraw) external;
 
    function setYHAFeesWithdraw(uint256[] memory _yHAFeesWithdraw) external;
 
    // ========================= HA ============================
 
    function getTotalCAmount() external view returns (uint256);
 
    function getPosition(uint256 vaultID, uint256 rate) external view returns (uint256, uint256);
 
    function createVault(
        address owner,
        uint256 broughtAmount,
        uint256 committedAmount
    ) external returns (uint256 vaultID);
 
    function cashOutVault(address caller, uint256 vaultID) external returns (uint256);
 
    function addToVault(
        address caller,
        uint256 vaultID,
        uint256 amount
    ) external;
 
    function removeFromVault(
        address caller,
        uint256 vaultID,
        uint256 amount
    ) external returns (uint256);
 
    // ========================= Keepers ===========================
 
    function tryCashOutVault(uint256 vaultID)
        external
        returns (
            address owner,
            uint256[3] memory // position, ratio, fees
        );
 
    function tryLiquidateVault(uint256 vaultID) external returns (uint256);
 
    function setFeeKeeper(uint256 feeDeposit, uint256 feesWithdraw) external;
}

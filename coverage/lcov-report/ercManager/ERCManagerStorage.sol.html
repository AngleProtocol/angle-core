pragma solidity 0.8.0;
 
import "@openzeppelin/contracts/token/ERC20/IERC20.sol";
import { IERC20 } from "@openzeppelin/contracts/token/ERC20/IERC20.sol";
import { SafeERC20 } from "@openzeppelin/contracts/token/ERC20/utils/SafeERC20.sol";
 
import "../external/PausableAgents.sol";
import "../external/FunctionUtils.sol";
 
import "../interfaces/IManager.sol";
import "../interfaces/ILendingAdapter.sol";
import "../interfaces/IVaultManager.sol";
import "../interfaces/IStableMaster.sol";
import "../interfaces/ISanToken.sol";
 
import "./ERCManagerEvents.sol";
 
/// @title Manager for ERC20 like collateral
/// @author Angle Core Team
/// @notice Manager of the ERC tokens.
contract ERCManagerStorage is ERCManagerEvents {
    // ============================ Lending ===================================
    //TODO lendingBounds need to hardcode coherent bounds such that there is not
    //     more than 100% lent or other issues
    struct LendingAdapter {
        ILendingAdapter adapter;
        uint256 lowerBound;
        uint256 upperBound;
        uint256 target;
        uint256 amountLent;
    }
 
    /// @notice mapping between the name of a lending platform and the corresponding details
    mapping(bytes32 => LendingAdapter) public lendingAdapterMap;
    /// @notice list of all the lending platforms
    bytes32[] public lendingAdapterList;
 
    /// @notice interface for the underlying token accepted by this contract
    IERC20 public token;
 
    /// @notice name of the collateral
    string public name;
 
    /// @notice reference to the stableMaster contract corresponding to this collateral manager
    IStableMaster public stableMaster;
 
    /// @notice reference to the sanToken contract corresponding to this collateral/stablecoin pair
    ISanToken public sanToken;
 
    /// @notice reference to the vaultManager for this collateral/stablecoin pair
    IVaultManager public vaultManager;
 
    /// @notice keeper contract for this collateral/stablecoin pair
    IKeeperFees public keeper;
 
    /// @notice base used for computing floats. It needs to be the same for all contracts
    uint256 public base;
 
    /// @notice portion of the fees that go to keepers liquidating HA vaults
    uint256 public keeperFeesLiquidate;
    // TODO change the cap and figure out the optimal value
    /// @notice cap on the fees that go to keepers liquidating vaults
    uint256 public keeperFeesCashOutCap;
 
    /// @notice thresholds on the values of the spread |gap leverage to target - committed|
    uint256[] public xKeeperFeesCashOut;
    /// @notice values of the portion of the fees that go to keepers in this case
    uint256[] public yKeeperFeesCashOut;
}

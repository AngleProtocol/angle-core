pragma solidity 0.8.0;
 
import './IVaultManager.sol';
import './IStableMaster.sol';
import './IManager.sol';
 
/// @title Interface of the contract managing vaults
/// @author Angle Core Team
/// @notice All the internal functions and variables are in the file parameters
/// @dev This contract does not interact directly with users
interface IKeeperFees {
 
	function init(address governor, address guardian) external;
 
	function updateUsersSLP() external;
 
	function updateHA() external;
 
	// ============================== Governance ======================================
 
	function setFeeMint(uint256[] memory _XFeeMint, uint256[] memory _YFeeMint) external;
 
	function setFeeBurn(uint256[] memory _XFeeBurn, uint256[] memory _YFeeBurn) external;
 
	function setSlippage(uint256[] memory _XSlippage, uint256[] memory _YSlippage) external;
 
	function setSlippageFee(uint256[] memory _XSlippageFee, uint256[] memory _YSlippageFee) external;
 
	function setHAFees(uint256 _HAFeeDeposit, uint256 _HAFeeWithdraw) external;
 
	function setGuardian(address _guardian, address guardian) external;
 
	function revokeGuardian(address guardian) external;
 
	function setStableMaster(IStableMaster _stableMaster) external;
 
	function setBase(uint256 _BASE) external;
}

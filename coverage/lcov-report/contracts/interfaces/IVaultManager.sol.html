pragma solidity 0.8.0;
 
import '@openzeppelin/contracts-upgradeable/access/AccessControlUpgradeable.sol';
 
import './IKeeperFees.sol';
import './IOracle.sol';
import './IStableMaster.sol';
import './IManager.sol';
 
/// @title Interface of the contract managing vaults
/// @author Angle Core Team
/// @notice All the internal functions and variables are in the file parameters
/// @dev This contract does not interact directly with users
interface IVaultManager is IAccessControlUpgradeable {
 
	// ========================= Events ====================
 
	event ModifiyingVault(
		address _owner,
		uint256 _init_rate,
		uint256 _broughtAmount,
		uint256 _committedAmount,
		uint256 _fees
	);
 
	event Timestamps(uint256 time1, uint256 time2);
 
	event PositionChange(uint256 position);
 
	event RewardAdded(uint256 reward);
 
	event RewardPaid(address indexed user, uint256 reward);
 
	// ========================= Governance ====================
 
	function deployCollateral(
		address governor,
		address guardian,
		IKeeperFees keeper
	) external;
 
	// temporary for testing the contract
	function setManagerRole(address tester) external;
 
	function setStableMaster(IStableMaster _stableMaster) external;
 
	function setKeeper(IKeeperFees _keeperFees) external;
 
	function setOracle(IOracle _oracle) external;
 
	function setBase(uint256 _BASE) external;
 
	function setSecureBlocks(uint256 _secureBlocks) external;
 
	function setMaxLeverage(uint256 maxLeverageNumerator, uint256 maxLeverageDenominator) external;
 
	function setMurDistribution(uint256 _rewardsDuration, address _rewardsDistribution) external;
 
	function deactivateMurDistribution() external;
 
	function setXHAFeesDeposit(uint256[] memory _XHAFeesDeposit) external;
 
	function setYHAFeesDeposit(uint256[] memory _YHAFeesDeposit) external;
 
	function setXHAFeesWithdraw(uint256[] memory _XHAFeesWithdraw) external;
 
	function setYHAFeesWithdraw(uint256[] memory _YHAFeesWithdraw) external;
 
	// ========================= HA ============================
 
	function createVault(
		address owner,
		uint256 broughtAmount,
		uint256 committedAmount
	) external;
 
	function cashOutVault(address owner) external returns (uint256);
 
	function addToVault(address owner, uint256 amount) external;
 
	function removeFromVault(address owner, uint256 amount) external returns (uint256);
 
	function mergeVaults(address source, address dest) external;
 
	function giveVault(address source, address dest) external;
 
    function getPosition(address owner, uint256 rate) external view returns (uint256, uint256);
 
	// ========================= Keepers ===========================
 
	function tryCashOutVault(address owner)
		external
		returns (
			uint256 position,
			uint256 ratio,
			uint256 fees
		);
 
	function tryLiquidateVault(address owner) external returns (uint256);
 
	function setFeeKeeper(uint256 feeDeposit, uint256 feesWithdraw) external;
 
	// ========================= Getters ============================
 
	function getTotalCAmount() external view returns (uint256);
}

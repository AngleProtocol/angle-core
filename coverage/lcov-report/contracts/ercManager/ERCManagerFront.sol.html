// SPDX-License-Identifier: UNLICENSED
 
pragma solidity 0.8.0;
 
import './ERCManager.sol';
 
/// @title ERCManagerFront
/// @author Angle Core Team
/// @notice Manager of the ERC tokens, front end (external functions).
contract ERCManagerFront is ERCManager {
	using SafeERC20 for IERC20;
 
	// ============================ Constructor ===================================
 
	/// @notice Constructor
	/// @param _name of the collateral
	/// @param _token address
	/// @param _stableMaster master stablecoin interface
	function initialize(
		string memory _name,
		address _token,
		IStableMaster _stableMaster
	) public initializer {
		initializePausableAgents();
		initializeParameters(_name, _token, _stableMaster);
		initializeRoles();
		//        _pauseStabler();
		//        _pauseHA();
	}
 
	// ============================ Users ===================================
 
	/// @notice Adds collateral to the system for minted stablecoins
	/// @param user address of the contract to mint to
	/// @param amount collateral sent
	/// @dev Amount is not scaled, so 1 means that if decimals = 18, it is 10**(-18) tokens
	function mint(address user, uint256 amount) external whenNotPausedStabler {
		// Important in this case (unless transfer reverts)
		require(int256(amount) > 0, 'ERCManagerFront::mint: Overflow');
		token.safeTransferFrom(msg.sender, address(this), amount);
		stableMaster.notifyMint(amount, user);
	}
 
	// ============================ SLP ===================================
 
	/// @notice Adds collateral to the system in exchange of sanTokens
	/// @param user address of the contract to send sanToken to
	/// @param amount of collateral sent
	function deposit(address user, uint256 amount) external whenNotPausedSLP {
		require(int256(amount) > 0, 'ERCManagerFront::deposit: Overflow');
		token.safeTransferFrom(msg.sender, address(this), amount);
		stableMaster.notifyDeposit(amount, user);
	}
 
	// ============================ HA ===================================
 
	/// @notice enter a ha (if no revert) for this stable coin and collateral
	/// @param amountBrought of collateral brought by the HA
	/// @param amountCommitted of collateral insured by the HA
	function joinHA(uint256 amountBrought, uint256 amountCommitted) external whenNotPausedHA returns (bool) {
		require(int256(amountBrought) > 0, 'ERCManagerFront::joinHA: Overflow bring');
		require(int256(amountCommitted) > 0, 'ERCManagerFront::joinHA: Overflow commit');
		token.safeTransferFrom(msg.sender, address(this), amountBrought);
		vaultManager.createVault(msg.sender, amountBrought, amountCommitted);
 
		emit JoiningHA(msg.sender, amountBrought, amountCommitted);
		return true;
	}
 
	/// @notice add amount brought by ha (if no revert) for this stablecoin and collateral
	/// @param amount to add to the vault
	function addHA(uint256 amount) external whenNotPausedHA returns (bool) {
		require(int256(amount) > 0, 'ERCManagerFront::addHA: Overflow');
		token.safeTransferFrom(msg.sender, address(this), amount);
		vaultManager.addToVault(msg.sender, amount);
 
		emit AddingHA(msg.sender, amount);
		return true;
	}
 
	/// @notice cash out an ha (if no revert) for this stable coin and collateral
	function exitHA() external whenNotPausedHA returns (bool) {
		uint256 amountReturn = vaultManager.cashOutVault(msg.sender);
		require(secureTransfer(msg.sender, amountReturn), 'ERCManagerFront::exitHA: Failed transfer');
 
		emit ExitingHA(msg.sender, amountReturn);
		return true;
	}
 
	/// @notice allow outside caller to liquidate a vault if ha vault is dead
	/// @param toLiquidate the address of the targeted vault
	function liquidateHA(address toLiquidate) external whenNotPausedHA returns (bool) {
		uint256 originalFees = vaultManager.tryLiquidateVault(toLiquidate);
 
		// compute the incentive for the keeper
		uint256 keeperFees = (originalFees * keeperFeesLiquidate) / BASE;
 
		require(secureTransfer(msg.sender, keeperFees), 'ERCManagerFront::liquidateHA: Failed transfer to keeper');
 
		emit LiquidatingHA(toLiquidate);
		emit KeeperFees(msg.sender, keeperFeesLiquidate, keeperFees, 0, originalFees);
		return true;
	}
 
	/// @notice allow outside caller to cash out a vault if the size leverage is too large
	/// @param toCashOut address of the targeted vault
	function cashOutHA(address toCashOut) external whenNotPausedHA returns (bool) {
		uint256 amountReturn;
		uint256 ratio;
		uint256 fees;
		(amountReturn, ratio, fees) = vaultManager.tryCashOutVault(toCashOut);
		require(secureTransfer(toCashOut, amountReturn), 'ERCManagerFront::cashOutHA: Failed transfer');
 
		uint256 rateFeeHA = FunctionUtils.piecewiseLinear(ratio, XKeeperFeesCashOut, YKeeperFeesCashOut);
		uint256 keeperFees = (fees * rateFeeHA) / BASE;
		uint256 keeperFeesCap = keeperFees < keeperFeesCashOutCap ? keeperFees : keeperFeesCashOutCap;
 
		require(secureTransfer(msg.sender, keeperFeesCap), 'ERCManagerFront::cashOutHA: Failed transfer to keeper');
 
		emit CashingOutHA(msg.sender, toCashOut, amountReturn, keeperFees);
		emit KeeperFees(msg.sender, rateFeeHA, keeperFeesCap, ratio, fees);
 
		return true;
	}
 
	/// @notice remove amount brought by ha for this stablecoin and collateral
	/// @param amount to remove
	function removeHA(uint256 amount) external whenNotPausedHA returns (bool) {
		require(int256(amount) > 0, 'ERCManagerFront::removeHA: Overflow');
		uint256 netAmount = vaultManager.removeFromVault(msg.sender, amount);
		require(secureTransfer(msg.sender, netAmount), 'ERCManagerFront::removeHA: Failed transfer');
 
		emit RemovingHA(msg.sender, amount, netAmount);
		return true;
	}
 
	/// @notice merge 2 vaults into dest vault (if no revert) for this stablecoin and collateral
	/// @param dest receiver address of the merge
	function mergeHA(address dest) external whenNotPausedHA returns (bool) {
		vaultManager.mergeVaults(msg.sender, dest);
		emit MergingHA(msg.sender, dest);
		return true;
	}
 
	/// @notice merge 2 vaults into dest vault (if no revert) for this stablecoin and collateral
	/// @param dest receiver address of the merge
	function giveHA(address dest) external whenNotPausedHA returns (bool) {
		vaultManager.giveVault(msg.sender, dest);
		emit GivingHA(msg.sender, dest);
		return true;
	}
 
	// ============================ Lending ===================================
 
	/// @notice correct lending rate for all lending protocol
	function correctLendingRate() external whenNotPausedSLP {
		uint256 gains = 0;
		for (uint256 i = 0; i < lendingInterfacesList.length; i++) {
			gains = gains + _correctLendingRate(lendingInterfacesList[i]);
		}
		if (gains > 0) {
			stableMaster.accumulateFees(gains);
			emit DistributingFees(gains);
		}
	}
 
	/// @notice correct lending rate for the associated protocol name
	/// @param platformName of the lending protocol
	function correctLendingRate(bytes32 platformName) external whenNotPausedSLP {
		// No need to check that it's a correct platformName here because 0 will be returned anyway
		uint256 gains = _correctLendingRate(platformName);
		if (gains > 0) {
			stableMaster.accumulateFees(gains);
			emit DistributingFees(gains);
		}
	}
 
	/// @notice accumulate the lending earnings from all lending protocols
	function distributeLendingGains() external whenNotPausedSLP {
		_distributeLendingGains();
	}
 
	/// @notice accumulate the lending earnings from a single protocol
	/// @param platformName of the lending protocol
	function distributeLendingGains(bytes32 platformName) external whenNotPausedSLP {
		// No need to check that it's a correct platformName here because 0 will be returned anyway
		_distributeLendingGains(platformName);
	}
}

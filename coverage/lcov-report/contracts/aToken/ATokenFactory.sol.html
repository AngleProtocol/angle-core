pragma solidity 0.8.0;
 
import './AToken.sol';
 
contract ATokenFactory {
	constructor() {}
 
	address public lastDeployed;
 
	function newAToken(
		string memory _name,
		string memory _symbol,
		IStableMaster _stableMaster
	) public {
		AToken tok = (new AToken(_name, _symbol, _stableMaster));
		lastDeployed = address(tok);
	}
}

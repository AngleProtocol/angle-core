pragma solidity 0.8.0;
 
import '../external/Decimal.sol';
 
import '../interfaces/IOracle.sol';
 
contract MockOracleFactory {
	constructor() {}
 
	address public lastDeployed;
 
	function newMockOracle(uint256 rate_) public {
		MockOracle c = (new MockOracle(rate_));
		lastDeployed = address(c);
	}
}
 
contract MockOracle is IOracle {
 
	uint256 BASE = 10**18;
	uint256 rate;
	bool public outdated;
 
	/// @notice Initiate with a fixe change rate
	constructor(uint256 rate_) {
		rate = rate_ * BASE / 100000;
	}
 
	/// @notice Mock update
	function update() external override returns (bool) {
		return true;
	}
 
	/// @notice Mock read
	function read() external view override returns (uint256, bool) {
		return (rate, true);
	}
 
	/// @notice change oracle rate
	function update(uint256 new_rate_) external override {
		rate = new_rate_ * BASE / 100000;
	}
 
	/// @notice Mock read
	function reader() external view returns (uint256, bool) {
		return (rate, true);
	}
 
	function lastUpdateTimestamp() external view override returns (uint256) {
		return block.timestamp;
	}
}

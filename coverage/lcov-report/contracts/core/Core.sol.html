pragma solidity 0.8.0;
 
import '@openzeppelin/contracts/access/AccessControl.sol';
import '@openzeppelin/contracts-upgradeable/proxy/utils/Initializable.sol';
 
import '../interfaces/ICore.sol';
 
/// @title Core contract
/// @author Angle Core Team
/// @notice Keeps track of all the stableMaster contracts and facilitates governance
contract Core is ICore, AccessControl, Initializable {
	bytes32 public constant GOVERNOR_ROLE = keccak256('GOVERNOR_ROLE');
	bytes32 public constant GUARDIAN_ROLE = keccak256('GUARDIAN_ROLE');
 
	/// @notice address of the governor for Angle's protocol
	address public governor;
 
	/// @notice address of the guardian key, it is going to be revoked
	address public guardian;
 
	/// @notice maps a name of a stablecoin to the address of the corresponding stablemaster
	mapping(string => IStableMaster) public mapName2Stable;
 
	/// @notice list of all the stablecoins accepted by the system
	string[] public stablecoinList;
 
	// ========================= Constructor ===================
 
	/// @notice core initializer
	/// @notice the core constructor
	/// @param admin address
	constructor(address admin) {
		_setupRole(GOVERNOR_ROLE, admin);
		_setRoleAdmin(GUARDIAN_ROLE, GOVERNOR_ROLE);
		_setRoleAdmin(GOVERNOR_ROLE, GOVERNOR_ROLE);
	}
 
	/// @notice core initializer
	/// @dev called by the admin address
	/// @param _governor address of the protocol
	/// @param _guardian address
	function initiateGovernance(address _governor, address _guardian)
		external
		override
		initializer
		onlyRole(GOVERNOR_ROLE)
	{
		grantRole(GOVERNOR_ROLE, _governor);
		grantRole(GUARDIAN_ROLE, _governor);
		grantRole(GUARDIAN_ROLE, _guardian);
		governor = _governor;
		guardian = _guardian;
	}
 
	// ========================= Governance =====================
 
	/// @notice to notify the address of a new stablemaster
	/// @param name of the stablecoin
	/// @param stableMaster of the stablecoin
	/// @param aToken of the stablecoin
	function deployStableMaster(
		string memory name,
		IStableMaster stableMaster,
		address aToken
	) external override onlyRole(GOVERNOR_ROLE) {
		stableMaster.init(governor, guardian, aToken);
 
		mapName2Stable[name] = stableMaster;
		stablecoinList.push(name);
	}
 
	// ========================= Access Control  =====================
 
	/// @notice to revoke the admin
	/// @param admin address
	function revokeAdmin(address admin) external override onlyRole(GOVERNOR_ROLE) {
		revokeRole(GOVERNOR_ROLE, admin);
	}
 
	/// @notice to change the governor address
	/// @param _governor new  address
	function setGovernor(address _governor) external override onlyRole(GOVERNOR_ROLE) {
		grantRole(GOVERNOR_ROLE, _governor);
		grantRole(GUARDIAN_ROLE, _governor);
		revokeRole(GUARDIAN_ROLE, governor);
		revokeRole(GOVERNOR_ROLE, governor);
 
		for (uint256 i = 0; i < stablecoinList.length; i++) {
			mapName2Stable[stablecoinList[i]].setGovernor(_governor);
		}
 
		governor = _governor;
	}
 
	/// @notice to change the guardian address
	/// @param _guardian new  address
	function setGuardian(address _guardian) external override onlyRole(GOVERNOR_ROLE) {
		grantRole(GUARDIAN_ROLE, _guardian);
		revokeRole(GUARDIAN_ROLE, guardian);
 
		for (uint256 i = 0; i < stablecoinList.length; i++) {
			mapName2Stable[stablecoinList[i]].setGuardian(_guardian);
		}
 
		guardian = _guardian;
	}
 
	/// @notice to revoke the guardian address
	function revokeGuardian() external override onlyRole(GOVERNOR_ROLE) {
		revokeRole(GUARDIAN_ROLE, guardian);
 
		for (uint256 i = 0; i < stablecoinList.length; i++) {
			mapName2Stable[stablecoinList[i]].revokeGuardian();
		}
	}
 
	// ========================= General Setters  =====================
 
	/// @notice porpagate a new stable master contract
	/// @param stable contract
	/// @param _stableMaster new contract
	function setStableMaster(string memory stable, IStableMaster _stableMaster)
		external
		override
		onlyRole(GOVERNOR_ROLE)
	{
		IStableMaster stablecoin = mapName2Stable[stable];
		if (address(stablecoin) != address(0)) {
			stablecoin.setStableMaster(_stableMaster);
		}
		mapName2Stable[stable] = _stableMaster;
	}
 
	/// @notice set a new BASE (used for proportion ratio and floats) for the whole system
	/// @param BASE new precision for decimals
	function setBase(uint256 BASE) external override onlyRole(GUARDIAN_ROLE) {
		for (uint256 i = 0; i < stablecoinList.length; i++) {
			mapName2Stable[stablecoinList[i]].setBaseGlobal(BASE);
		}
	}
}

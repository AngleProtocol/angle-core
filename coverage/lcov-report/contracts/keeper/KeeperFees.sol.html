pragma solidity 0.8.0;
 
import '@openzeppelin/contracts/access/AccessControl.sol';
import '@openzeppelin/contracts-upgradeable/proxy/utils/Initializable.sol';
 
import '../external/FunctionUtils.sol';
 
import '../interfaces/IKeeperFees.sol';
 
import './KeeperFeesStorage.sol';
 
/// @title Contract for keeper utility functions
/// @author Angle Core Team
/// @dev This contract interacts with parameters for a given stablecoin/collateral pair
contract KeeperFees is IKeeperFees, KeeperFeesStorage, AccessControl, Initializable {
	bytes32 public constant MANAGER_ROLE = keccak256('MANAGER_ROLE');
	// when private key is not needed anymore the gov should be able to revoke the guardian
	bytes32 public constant GUARDIAN_ROLE = keccak256('GUARDIAN_ROLE');
 
	/// @notice deploy the keeper utils function for this (stable,collateral)
	/// @param _stableMaster contract
	/// @param _manager contract
	/// @param _vaultManager contract
	constructor(
		IStableMaster _stableMaster,
		IManager _manager,
		IVaultManager _vaultManager
	) KeeperFeesStorage(_stableMaster, _manager, _vaultManager) {
		_setupRole(MANAGER_ROLE, address(_manager));
 
		_setRoleAdmin(MANAGER_ROLE, MANAGER_ROLE);
		_setRoleAdmin(GUARDIAN_ROLE, MANAGER_ROLE);
	}
 
	/// @notice the keeperFees init to grant roles
	/// @param governor address of the protocol
	/// @param guardian address
	function init(address governor, address guardian) external override onlyRole(MANAGER_ROLE) initializer {
		grantRole(GUARDIAN_ROLE, governor);
		grantRole(GUARDIAN_ROLE, guardian);
	}
 
	// ============================ StableMaster ===================================
 
	/// @notice Update the SLP and Users fees associated to the (stable, collat). Are updated:
	/// 	-	Mint Fee (fee induced by a user minting)
	///		-	Burn Fee (fee induced by a user burning)
	///		-	Slippage (what's taken by SLPs compared with what they want)
	///		-	SlippageFee that is the part of fees that's put aside because the protocol is not well collateralized
	function updateUsersSLP() external override {
		uint256 collatRatio = stableMaster.getCollateralRatio();
		uint256 feeBurn = FunctionUtils.piecewiseLinear(collatRatio, XFeeBurn, YFeeBurn);
		uint256 feeMint = FunctionUtils.piecewiseLinear(collatRatio, XFeeMint, YFeeMint);
		uint256 slippage = FunctionUtils.piecewiseLinear(collatRatio, XSlippage, YSlippage);
		uint256 slippageFee = FunctionUtils.piecewiseLinear(collatRatio, XSlippageFee, YSlippageFee);
 
		stableMaster.setFeeKeeper(manager, feeMint, feeBurn, slippage, slippageFee);
 
		emit userAndSLPFeesUpdated(collatRatio,feeBurn,feeMint,slippage,slippageFee);
	}
 
	// ============================= VaultManager =================================
 
	/// @notice Update HA fees associated to the (stable, collat) pair
	function updateHA() external override {
		vaultManager.setFeeKeeper(HAFeeDeposit, HAFeeWithdraw);
 
		emit HAFeesUpdated(HAFeeDeposit,HAFeeWithdraw);
	}
 
	// ============================= Governance =================================
 
	/// @notice set the x(ie thresholds) / y(ie values)-array  fees for minting
	/// @param _XFeeMint new collateral ratio thresholds
	/// @param _YFeeMint new mint fees at thresholds
	function setFeeMint(
		uint256[] memory _XFeeMint, 
		uint256[] memory _YFeeMint
	)
		external
		override
		onlyRole(GUARDIAN_ROLE)
	{
		XFeeMint = _XFeeMint;
		YFeeMint = _YFeeMint;
	}
 
	/// @notice set the x/y-array (ie thresholds) fees for burning
	/// @param _XFeeBurn new collateral ratio thresholds
	/// @param _YFeeBurn new burn fees at thresholds
	function setFeeBurn(
		uint256[] memory _XFeeBurn, 
		uint256[] memory _YFeeBurn
	)
		external
		override
		onlyRole(GUARDIAN_ROLE)
	{
		XFeeBurn = _XFeeBurn;
		YFeeBurn = _YFeeBurn;
	}
 
	/// @notice set the x(ie thresholds) / y(ie values) for SLP to withdraw from the protocol
	/// @param _XSlippage new collateral ratio thresholds
	/// @param _YSlippage new slippage at threshold
	function setSlippage(
		uint256[] memory _XSlippage, 
		uint256[] memory _YSlippage
	)
		external
		override
		onlyRole(GUARDIAN_ROLE)
	{
		XSlippage = _XSlippage;
		YSlippage = _YSlippage;
	}
 
	/// @notice set the x(ie thresholds) / y(ie values)-array for the fees withdrawn
	///         from SLP depending on the col ratio
	/// @param _XSlippageFee new collateral ratio thresholds
	/// @param _YSlippageFee new slippageFee at thresholds
	function setSlippageFee(
		uint256[] memory _XSlippageFee, 
		uint256[] memory _YSlippageFee
	)
		external
		override
		onlyRole(GUARDIAN_ROLE)
	{
		XSlippageFee = _XSlippageFee;
		YSlippageFee = _YSlippageFee;
	}
 
	/// @notice set the extra fees that can be used when HAs mint and burn
	/// @param _HAFeeDeposit new deposit fee for HAs
	/// @param _HAFeeWithdraw new withdraw fee for HAs
	function setHAFees(
		uint256 _HAFeeDeposit, 
		uint256 _HAFeeWithdraw
	) 
		external 
		override 
		onlyRole(GUARDIAN_ROLE)
	{
		HAFeeDeposit = _HAFeeDeposit;
		HAFeeWithdraw = _HAFeeWithdraw;
	}
 
	// =========================== Governance / Setters  ================================
 
	/// @notice to change the guardian address
	/// @param _guardian new address
	/// @param guardian old address
	function setGuardian(address _guardian, address guardian) external override onlyRole(MANAGER_ROLE) {
		revokeRole(GUARDIAN_ROLE, guardian);
		grantRole(GUARDIAN_ROLE, _guardian);
	}
 
	/// @notice to revoke the guardian address
	/// @param guardian guardian address to revoke
	function revokeGuardian(address guardian) external override onlyRole(MANAGER_ROLE) {
		revokeRole(GUARDIAN_ROLE, guardian);
	}
 
	/// @notice set a new stable master contract
	/// @param _stableMaster new contract
	function setStableMaster(IStableMaster _stableMaster) external override onlyRole(MANAGER_ROLE) {
		stableMaster = _stableMaster;
	}
 
	/// @notice allows to propagate the change of base
	/// @param _BASE new BASE
	function setBase(uint256 _BASE) external override onlyRole(MANAGER_ROLE) {
		BASE = _BASE;
	}
}

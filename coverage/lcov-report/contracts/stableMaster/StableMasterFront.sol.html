// SPDX-License-Identifier: UNLICENSED
 
pragma solidity 0.8.0;
 
import './StableMaster.sol';
 
/// @title StableMasterFront
/// @author Angle Core Team
/// @notice All the functions that can be called by external keepers
/// @dev This contract contains the only non-view functions that can be called by other agents
contract StableMasterFront is StableMaster {
 
	/// @notice Initializes the name of the stablecoin
	/// @param core address
	/// @param _name of the stableMaster
	constructor(address core, string memory _name) StableMaster(core, _name) {}
 
	// ============================== SLP ======================================
 
	/// @notice Update the sanRate for a collateral type
	/// @param manager contract of the collateral requested
	function updateSanRate(IManager manager) external {
		// Search collateral data
		Collateral storage col = collateralMap[manager];
		// No need to check if the collateral is a correct collateral
 
		// Compute fraction of fee aside if colRate is small enough
		uint256 toShare = col.feesAccumulated;
		col.feesAccumulated = 0;
		if ((col.slippageFee == 0) && (col.feesAside != 0)) {
			toShare = toShare + col.feesAside;
			col.feesAside = 0;
		} else {
			// Rounding above
			uint256 aside = toShare - ((toShare * (BASE - col.slippageFee)) / BASE);
			toShare = toShare - aside;
			col.feesAside = col.feesAside + aside;
		}
		uint256 sanMint = getSanMint(manager);
		if (sanMint != 0) {
			uint256 oldSanRate = col.sanRate;
			col.sanRate = oldSanRate + toShare * BASE / sanMint;
			SanRateUpdated(oldSanRate, col.sanRate);
		}
	}
 
}

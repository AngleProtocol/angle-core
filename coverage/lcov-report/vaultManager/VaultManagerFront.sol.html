pragma solidity 0.8.2;
 
import "./VaultManager.sol";
 
/// @title VaultManagerFront
/// @author Angle Core Team
/// @notice `VaultManager` is the contract handling all the Hedging Agents cashOutAmounts and vaults
/// @dev There is one `VaultManager` contract per pair stablecoin/collateral in the protocol
/// @dev This file contains the functions of the `VaultManager` that can be directly interacted with by external agents
/// @dev `VaultManager` naturally handles staking, the code allowing HAs to stake has been inspired from
/// https://github.com/SetProtocol/index-coop-contracts/blob/master/contracts/staking/StakingRewardsV2.sol
contract VaultManagerFront is IVaultManagerFront, VaultManager, ReentrancyGuardUpgradeable {
    // ============================= Deployer =================================
 
    /// @notice Initializes the `VaultManager` contract
    /// @param _stableMaster Reference to the stableMaster contract for this stablecoin
    /// @param _manager Reference to the manager contract handling the collateral associated to the `VaultManager`
    /// @param _oracle Reference to the oracle contract that will give the price of the collateral with respect to the stablecoin
    /// It will be used to compute HA's cashOutAmounts
    /// @param _rewardToken Reference to the rewardtoken that can be distributed to HAs as they have open cashOutAmounts
    /// @dev The reward token is most likely going to be the MUR token
    /// @dev Since this contract is upgradable this function is an initialize and not a constructor
    function initialize(
        IStableMaster _stableMaster,
        IManager _manager,
        IOracle _oracle,
        IERC20 _rewardToken
    ) external initializer {
        // Initializing contracts
        __ReentrancyGuard_init();
        __Pausable_init();
        __AccessControl_init();
 
        // Creating references
        stableMaster = _stableMaster;
        oracle = _oracle;
        manager = _manager;
        rewardToken = _rewardToken;
 
        // Initializing the fees and parameters of this contract
        // Governance can update these values afterwards
        // This structure of fees was chosen to facilitate testing
        collatBase = oracle.getInBase();
        maxLeverage = 3 * BASE;
        cashOutLeverage = 100 * BASE;
        maintenanceMargin = (3 * BASE) / 1000;
        xHAFeesDeposit = [(3 * BASE) / 10, (6 * BASE) / 10, (9 * BASE) / 10];
        // Values in the array above should normally be increasing: the lower the x the cheaper it should
        // be for HAs to come in
        yHAFeesDeposit = [BASE / 100, BASE / 100, BASE / 100];
        haFeesDepositKeeper = BASE;
        xHAFeesWithdraw = [(3 * BASE) / 10, (6 * BASE) / 10, (9 * BASE) / 10];
        // Values in the array above should normally be decreasing: the higher the x the cheaper it should
        // be for HAs to come in
        yHAFeesWithdraw = [BASE / 100, BASE / 100, BASE / 100, BASE / 100];
        haFeesWithdrawKeeper = BASE;
 
        // Setting up access control for this contract
        _setupRole(MANAGER_ROLE, address(manager));
        // Manager is admin of all the roles, for which changes are to be propagated from it
        _setRoleAdmin(GUARDIAN_ROLE, MANAGER_ROLE);
        _setRoleAdmin(KEEPER_ROLE, MANAGER_ROLE);
        _setRoleAdmin(MANAGER_ROLE, MANAGER_ROLE);
    }
 
    // ============================= External View Functions =================================
 
    /// @notice Wrapper from `_getNetAmountAndFeesCreation`. Gets the value of the fees and the net amount
    /// corrected from the fees in case of vault creation
    /// @param committedAmount commited amount in the vault
    /// @param broughtAmount the brought amount in the vault
    /// @return netBroughtAmount Amount that will be written in the vault as the cashOutAmount
    /// @return fees Fees induced by this transaction
    function getNetAmountAndFeesCreation(uint256 broughtAmount, uint256 committedAmount)
        external
        view
        returns (uint256 netBroughtAmount, uint256 fees)
    {
        // Computing the oracle value
        // Only the highest oracle value (between Chainlink and Uniswap) we get is stored in the vault
        uint256 rateUp;
        (, rateUp) = _getOraclePrice();
        (netBroughtAmount, fees) = _getNetAmountAndFeesCreation(broughtAmount, committedAmount, rateUp);
    }
 
    /// @notice Wrapper from `_getNetAmountAndFeesUpdate`. Gets the value of the fees and the net amount corrected
    /// from the fees in case of a deposit or a withdraw of collateral from a HA vault
    /// @param amount Amount to deposit or withdraw from the vault
    /// @param entry Checker to see if the transaction is a deposit or a withdraw
    /// @return netAmount Amount that will be either written in the vault as the cashOutAmount or given back to the HA
    /// @return fees Fees induced by this transaction
    function getNetAmountAndFeesUpdate(uint256 amount, uint256 entry)
        external
        view
        returns (uint256 netAmount, uint256 fees)
    {
        // Computing the oracle value
        // Only the highest oracle value (between Chainlink and Uniswap) we get is stored in the vault
        uint256 rateUp;
        (, rateUp) = _getOraclePrice();
        (netAmount, fees) = _getNetAmountAndFeesUpdate(amount, rateUp, entry);
    }
 
    /// @notice Checks if HAs are covering too much collateral with respect to what they can cover
    /// @param amount Amount to add to the current total amount committed
    /// @param withdraw whether we should withdraw (or add) the amount from totalCAmount
    /// @return The new potential collateral insured amount (after taking into account amount)
    /// @return The maximum amount of collateral that can be insured by HAs
    /// @dev If amount = 0, then it is just a simple check on the current coverage ratio
    function testMaxCAmount(uint256 amount, uint256 withdraw) external view returns (uint256, uint256) {
        uint256 rateUp;
        (, rateUp) = _getOraclePrice();
        return _testMaxCAmount(amount, withdraw, rateUp);
    }
 
    /// @notice Returns the cashOutAmount of the vault owned by someone at a given oracle value
    /// @param vaultID ID of the vault
    /// @param rate Oracle value
    function getCashOutAmount(uint256 vaultID, uint256 rate) external view override returns (uint256, uint256) {
        return _getCashOutAmount(vaultID, rate);
    }
 
    /// @notice Helper function to check if the vault can be liquidated
    /// @param vaultID ID of the vault to liquidate
    /// @return isliquidable Whether the vault is liquidable
    /// @return fees The fees in this vault
    /// @dev This view function can be queried by keeper bots to maintain the protocol healthy
    /// @dev Combining this function with the computeKeeperHAFees of the `ERCManagerFront` can help keepers
    /// see if they would make a profit by cashing out the vault
    function testLiquidation(uint256 vaultID) external view returns (uint256 isliquidable, uint256 fees) {
        (uint256 rateDown, ) = _getOraclePrice();
        // To compute the cashOutAmount in this case, we use the lowest oracle value possible
        (uint256 cashOutAmount, uint256 reachMaintenanceMargin) = _getCashOutAmount(vaultID, rateDown);
        // Checking if the vault is liquidable - hitted or went below the maintenance margin
        isliquidable = cashOutAmount == 0 || reachMaintenanceMargin == 1 ? 1 : 0;
        fees = vaultData[vaultID].fees;
    }
 
    /// @notice Helper function to check if the vault can be cashed out due to a too high leverage or to the
    /// fact that too much collateral is covered by HAs
    /// @param vaultID ID of the vault
    /// @return canBeCashedOut Whether the vault can be cashed out
    /// @return ratioCoverVsTooMuchCovered The ratio between what's covered by the vault and the gap between what's
    /// covered and what's to cover. If this ratio is zero this means that HAs do not cover more than what they should
    /// cover
    /// @return fees Fees initially paid by the vault owner at creation
    /// @dev This view function can be queried by keeper bots to maintain the protocol healthy
    /// @dev Combining this function with the computeKeeperHAFees of the `ERCManagerFront` can help keepers
    /// see if they would make a profit by cashing out the vault
    function testCashOut(uint256 vaultID)
        external
        view
        returns (
            uint256 canBeCashedOut,
            uint256 ratioCoverVsTooMuchCovered,
            uint256 fees
        )
    {
        (uint256 rateDown, uint256 rateUp) = _getOraclePrice();
        (canBeCashedOut, , ratioCoverVsTooMuchCovered, , fees) = _getForcedExitInfo(vaultID, rateDown, rateUp);
    }

        _getReward(vaultID);
    }

    /// @notice Gets the balance of an owner
    /// @param owner Address of the owner
    /// @dev Balance here represents the number of vaults owned by a HA
    function balanceOf(address owner) public view override returns (uint256) {
        require(owner != address(0), "VaultManager::balanceOf: balance query for the zero address");
        return balances[owner];
    }
 
    /// @notice Gets the owner of the vault with ID vaultID
    /// @param vaultID ID of the vault
    function ownerOf(uint256 vaultID) public view override returns (address) {
        return _ownerOf(vaultID);
    }
 
    /// @notice Approves to an address specified by to on a vault specified by vaultID
    /// @param to Address to approve the vault to
    /// @param vaultID ID of the vault
    /// @dev The approved address will have the right to transfer the vault, or to cash it out
    /// on behalf of the owner
    function approve(address to, uint256 vaultID) public override {
        address owner = ownerOf(vaultID);
        require(to != owner, "VaultManager::approve: approval to current owner");
        require(
            msg.sender == owner || isApprovedForAll(owner, msg.sender),
            "VaultManager::approve: approve caller is not owner nor approved for all"
        );
 
        _approve(to, vaultID);
    }
 
    /// @notice Gets the approved address by a vault owner
    /// @param vaultID ID of the concerned vault
    function getApproved(uint256 vaultID) public view override returns (address) {
        return _getApproved(vaultID);
    }
 
    /// @notice Sets approval on all vaults owned by the owner to an operator
    /// @param operator Address to approve (or block) on all vaults
    /// @param approved Whether the sender wans to approve or block the operator
    function setApprovalForAll(address operator, bool approved) public override {
        require(operator != msg.sender, "VaultManager::setApprovalForAll: approve to caller");
        operatorApprovals[msg.sender][operator] = approved;
        emit ApprovalForAll(msg.sender, operator, approved);
    }
 
    /// @notice Gets if the operator address is approved on all vaults by the owner
    /// @param owner Owner of vaults
    /// @param operator Address to check if approved
    function isApprovedForAll(address owner, address operator) public view override returns (bool) {
        return operatorApprovals[owner][operator];
    }
 
    /// @notice Transfers the vaultID from an address to another

    // =========================== ERC165 logic ============================
 
    /// @notice Queries if a contract implements an interface
    /// @param interfaceId The interface identifier, as specified in ERC-165
    /// @dev Interface identification is specified in ERC-165. This function uses less than 30,000 gas.
    /// @return `true` if the contract implements `interfaceID` and
    ///  `interfaceID` is not 0xffffffff, `false` otherwise
    function supportsInterface(bytes4 interfaceId) public pure override(IERC165Upgradeable) returns (bool) {
        return
            interfaceId == type(IAccessControlUpgradeable).interfaceId ||
            interfaceId == type(IERC721Upgradeable).interfaceId ||
            interfaceId == type(IERC165Upgradeable).interfaceId;
    }

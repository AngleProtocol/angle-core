pragma solidity 0.8.0;
 
import "../external/Decimal.sol";
 
/// @title VaultManagerEvents
/// @author Angle Core Team
/// @notice All the events used in VaultManager contract
contract VaultManagerEvents {
    event Transfer(address indexed from, address indexed to, uint256 indexed tokenId);
 
    event Approval(address indexed owner, address indexed approved, uint256 indexed tokenId);
 
    event ApprovalForAll(address indexed owner, address indexed operator, bool approved);
 
    event CashingOut(uint256 vaultID, uint256 committedAmount, uint256 position);
 
    event Liquidating(uint256 vaultID, uint256 broughtAmount, uint256 previousAmountLock, uint256 totalCAmount);
 
    event TotalLeverageChange(
        int256 stockUsers,
        uint256 stablecoinsInCol,
        bool cAmountIsValid,
        uint256 newCAmount,
        uint256 maxCAmount
    );
 
    event UpdatingVault(
        uint256 vaultID,
        uint256 initRate,
        uint256 broughtAmount,
        uint256 committedAmount,
        uint256 fees
    );
 
    event ModifyingVault(
        uint256 vaultID,
        uint256 initRate,
        uint256 broughtAmount,
        uint256 committedAmount,
        uint256 fees
    );
 
    event Timestamps(uint256 time1, uint256 time2);
 
    event PositionChange(uint256 position);
 
    event RewardAdded(uint256 reward);
 
    event RewardPaid(address indexed user, uint256 reward);
}
